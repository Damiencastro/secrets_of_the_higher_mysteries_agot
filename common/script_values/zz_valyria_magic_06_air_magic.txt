zz_valyria_air_magic_stat_gui = {
	if = {
		limit = {
			exists = this 
		}
		value = "has_trait_xp(zz_magister|air_magic)" 
	}
	if = {
		limit = {
			has_variable = air_magic_overcap 
		}
		add = this.var:air_magic_overcap
	}
}
### Air Magic 
zz_valyria_magic_air_magic_base_value = { 
	scope:secondary_recipient = { 
		if = {
			limit = {
				exists = var:casttype
			}
			if = {
				limit = { 
					var:casttype ?= flag:potion 
				}
				value =  var:cast_potency
			}
			else_if = {
				limit = {
					var:casttype ?= flag:scroll
				}
				value =  var:cast_potency
			}
			else = { 
				value = "has_trait_xp(zz_magister|air_magic)" 
			}
		}
		else = { 
			value = "has_trait_xp(zz_magister|air_magic)" 
		}
		switch = {
			trigger = global_var:magic_level
			0 = {
				multiply = 0.1 
			}
			1 = {
				multiply = 0.5
			}
			2 = {
				multiply = 1
			}
		}
	}
}
zz_valyria_magic_air_magic_01_value = { 
	value =1 
	add = {
		scope:secondary_recipient = {
			value = zz_valyria_magic_air_magic_base_value
		}
	}
}
