zz_valyria_alchemy_magic_stat_gui = {
	if = {
		limit = {
			exists = this 
		}
		value = "has_trait_xp(zz_magister|alchemy_magic)" 
	}
	if = {
		limit = {
			has_variable = alchemy_magic_overcap 
		}
		add = this.var:alchemy_magic_overcap
	}
}
### Alchemy Magic
zz_valyria_magic_alchemy_magic_base_value = { 
	scope:secondary_recipient = { 
		if = {
			limit = {
				exists = var:casttype
			}
			if = {
				limit = { 
					var:casttype ?= flag:potion 
				}
				value =  var:cast_potency
			}
			else_if = {
				limit = {
					var:casttype ?= flag:scroll
				}
				value =  var:cast_potency
			}
			else = { 
				value = "has_trait_xp(zz_magister|alchemy_magic)" 
			}
		}
		else = { 
			value = "has_trait_xp(zz_magister|alchemy_magic)" 
		}
		switch = {
			trigger = global_var:magic_level
			0 = {
				multiply = 0.1 
			}
			1 = {
				multiply = 0.5
			}
			2 = {
				multiply = 1
			}
		}
	}
}
zz_valyria_magic_alchemy_magic_01_value = {
	value = 0
	add = { 
		scope:secondary_recipient = { 
			value = zz_valyria_magic_alchemy_magic_base_value
			divide = 10
			min = 1
		}
	}
}
