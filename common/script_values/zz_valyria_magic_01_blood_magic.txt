zz_valyria_blood_magic_stat_gui = {
	if = {
		limit = {
			exists = this 
		}
		value = "has_trait_xp(zz_magister|blood_magic)" 
	}
	if = {
		limit = {
			has_variable = blood_magic_overcap 
		}
		add = this.var:blood_magic_overcap
	}
}
### Blood Magic
zz_valyria_magic_blood_magic_base_value = { 
	scope:secondary_recipient = { 
		if = {
			limit = {
				exists = var:casttype
			}
			if = {
				limit = { 
					var:casttype ?= flag:potion 
				}
				value = scope:actor.var:cast_potency
			}
			else_if = {
				limit = {
					var:casttype ?= flag:scroll
				}
				value =  scope:actor.var:cast_potency
			}
			else = { 
				value = "has_trait_xp(zz_magister|blood_magic)" 
			}
		}
		else = { 
			value = "has_trait_xp(zz_magister|blood_magic)" 
		}
		switch = {
			trigger = global_var:magic_level
			0 = {
				multiply = 0.1 
			}
			1 = {
				multiply = 0.5
			}
			2 = {
				multiply = 1
			}
		}
	}
}
zz_valyria_magic_blood_magic_01_value = { 
	value = 0
	add = { 
		scope:secondary_recipient = { 
			value = zz_valyria_magic_blood_magic_base_value
			divide = 20
			min = 1
		}
	}
}
