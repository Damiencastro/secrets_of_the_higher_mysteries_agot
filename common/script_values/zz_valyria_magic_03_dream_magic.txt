zz_valyria_dream_magic_stat_gui = {
	if = {
		limit = {
			exists = this 
		}
		value = "has_trait_xp(zz_magister|dream_magic)" 
	}
	if = {
		limit = {
			has_variable = dream_magic_overcap 
		}
		add = this.var:dream_magic_overcap
	}
}
### Dream Magic!
zz_valyria_magic_dream_magic_base_value = { 
	scope:secondary_recipient = { 
		if = {
			limit = {
				exists = var:casttype
			}
			if = {
				limit = { 
					var:casttype ?= flag:potion 
				}
				value =  var:cast_potency
			}
			else_if = {
				limit = {
					var:casttype ?= flag:scroll
				}
				value =  var:cast_potency
			}
			else = { 
				value = "has_trait_xp(zz_magister|dream_magic)" 
				if = { 
					limit = {
						OR = {
							has_trait = dragon_dreams
							has_trait = greensight
						}
						
					}
					add = 50
				}
			}
		}
		else = { 
			value = "has_trait_xp(zz_magister|dream_magic)" 
			if = { 
				limit = {
					has_trait = dragon_dreams
				}
				add = 25
			}
			if = { 
				limit = {
					has_trait = greensight
				}
				add = 25
			}
		}
		switch = {
			trigger = global_var:magic_level
			0 = {
				multiply = 0.1 
			}
			1 = {
				multiply = 0.5
			}
			2 = {
				multiply = 1
			}
		}
	}
}
zz_valyria_magic_dream_magic_02_value = {
	value = 0
	add = { 
		scope:secondary_recipient = { 
			value = zz_valyria_magic_dream_magic_base_value
			divide = 2
			min = 25
		}
	}
}
