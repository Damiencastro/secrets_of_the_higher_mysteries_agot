zz_valyria_dragon_magic_stat_gui = {
	if = {
		limit = {
			exists = this 
		}
		value = "has_trait_xp(zz_magister|dragon_magic)" 
	}
	if = {
		limit = {
			has_variable = dragon_magic_overcap 
		}
		add = this.var:dragon_magic_overcap
	}
}
zz_valyria_magic_dragon_magic_base_value = { 
	scope:secondary_recipient = { 
		if = {
			limit = {
				exists = var:casttype
			}
			if = {
				limit = { 
					var:casttype ?= flag:potion 
				}
				value = scope:actor.var:cast_potency
			}
			else_if = {
				limit = {
					var:casttype ?= flag:scroll
				}
				value =  scope:actor.var:cast_potency
			}
			else = { 
				value = "has_trait_xp(zz_magister|dragon_magic)" 
			}
		}
		else = { 
			value = "has_trait_xp(zz_magister|dragon_magic)" 
		}
	}
}

	value = 0
	add = { 
		scope:secondary_recipient = { 
			value = zz_valyria_magic_dragon_magic_base_value
			min = 1
		}
	}
